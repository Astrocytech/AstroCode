import os
from functools import wraps
import time

def timer_decorator(func):
    @wraps(func)
    def wrapper_timer(*args, **kwargs):
        start_time = time.time()
        value = func(*args, **kwargs)
        end_time = time.time()
        print(f'Elapsed time: {end_time - start_time} seconds')
        return value
    return wrapper_timer
